-- 1. EXTENSOES
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 2. FUNCAO AUXILIAR PARA ATUALIZAR TIMESTAMP UPDATED_AT
CREATE OR REPLACE FUNCTION public.trigger_set_timestamp()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 3. ORGANIZACOES (Multi-Tenant)
CREATE TABLE IF NOT EXISTS public.organizations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name TEXT NOT NULL,
    type TEXT NOT NULL DEFAULT 'producer' CHECK (type IN ('producer', 'agronomy_consulting', 'cooperative', 'reseller', 'agrogb')),
    owner_user_id UUID,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

ALTER TABLE public.organizations ENABLE ROW LEVEL SECURITY;

CREATE TRIGGER set_timestamp_organizations
    BEFORE UPDATE ON public.organizations
    FOR EACH ROW EXECUTE FUNCTION public.trigger_set_timestamp();

-- 4. PERFIS DE USUARIOS (Profiles)
CREATE TABLE IF NOT EXISTS public.profiles (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    nome_completo TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    telefone TEXT,
    role TEXT NOT NULL DEFAULT 'CLIENTE' CHECK (role IN ('ADMIN', 'AGRONOMO', 'CLIENTE', 'STAFF')),
    status TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'inactive', 'suspended')),
    subscription_plan TEXT NOT NULL DEFAULT 'FREE',
    organization_id UUID REFERENCES public.organizations(id) ON DELETE SET NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

CREATE TRIGGER set_timestamp_profiles
    BEFORE UPDATE ON public.profiles
    FOR EACH ROW EXECUTE FUNCTION public.trigger_set_timestamp();

-- Column update
ALTER TABLE public.organizations ADD CONSTRAINT fk_owner_user FOREIGN KEY (owner_user_id) REFERENCES public.profiles(id) ON DELETE SET NULL;

-- 19. POLITICAS DE SEGURANCA (Row-Level Security)
-- 19.1 Profiles Policies
CREATE POLICY "Perfis visiveis por membros da mesma organizacao ou administradores"
    ON public.profiles FOR ALL
    USING (
        auth.uid() = id OR
        role = 'ADMIN' OR
        organization_id = (SELECT organization_id FROM public.profiles WHERE id = auth.uid())
    );

-- 19.2 Farms Policies
CREATE POLICY "Propriedades visiveis por donos, agronomos vinculados ou admins"
    ON public.farms FOR ALL
    USING (
        owner_id = auth.uid() OR
        created_by = auth.uid() OR
        organization_id = (SELECT organization_id FROM public.profiles WHERE id = auth.uid()) OR
        EXISTS (
            SELECT 1 FROM public.agronomist_client_links
            WHERE agronomist_id = auth.uid() AND client_id = owner_id AND status = 'ACTIVE'
        )
    );

-- 19.3 Fields Policies
CREATE POLICY "Talhoes visiveis por donos, agronomos vinculados ou admins"
    ON public.fields FOR ALL
    USING (
        farm_uuid IN (SELECT uuid FROM public.farms)
    );

-- 19.4 Plantings Policies
CREATE POLICY "Plantios visiveis por donos, agronomos vinculados ou admins"
    ON public.plantings FOR ALL
    USING (
        field_uuid IN (SELECT uuid FROM public.fields)
    );

-- 19.5 Recommendations Policies
CREATE POLICY "Recomendacoes visiveis por clientes e agronomos autores ou admins"
    ON public.recommendations FOR ALL
    USING (
        client_id = auth.uid() OR
        agronomist_id = auth.uid() OR
        organization_id = (SELECT organization_id FROM public.profiles WHERE id = auth.uid())
    );

-- 19.6 Products Policies (Global/Public Read, Admin Write)
CREATE POLICY "Insumos visiveis por todos os autenticados"
    ON public.products FOR SELECT
    USING (true);

CREATE POLICY "Insumos manipulaveis por donos ou admins"
    ON public.products FOR ALL
    USING (
        owner_id = auth.uid() OR
        (SELECT role FROM public.profiles WHERE id = auth.uid()) = 'ADMIN'
    );
