SELECT p.proname, r.rolname, has_function_privilege(r.rolname, p.oid, 'EXECUTE') FROM pg_proc p JOIN pg_namespace n ON n.oid = p.pronamespace CROSS JOIN pg_roles r WHERE n.nspname = 'public' AND p.proname IN ('apply_fertilization_v2', 'get_my_financial_summary', 'monitoramento_entidade_run_checks', 'process_sale_v2', 'handle_new_user', 'update_last_updated_column') AND r.rolname IN ('authenticated', 'anon') ORDER BY p.proname, r.rolname;
