-- Função: Otimização Extrema. Injeta a profissão (role) e a organização do 
-- usuário diretamente no Token de Login para que o RLS não precise fazer queries.
-- ==============================================================================

-- 1. FUNÇÕES UTILITÁRIAS PARA O RLS LER O JWT DIRETAMENTE O(1)
CREATE OR REPLACE FUNCTION auth.jwt_role() RETURNS TEXT AS $$
  SELECT NULLIF(current_setting('request.jwt.claims', true)::json->'app_metadata'->>'role', '')::TEXT;
$$ LANGUAGE SQL STABLE;

CREATE OR REPLACE FUNCTION auth.jwt_org() RETURNS UUID AS $$
  SELECT NULLIF(current_setting('request.jwt.claims', true)::json->'app_metadata'->>'organization_id', '')::UUID;
$$ LANGUAGE SQL STABLE;

-- 2. CRIAR A FUNÇÃO DO HOOK PARA O SUPABASE AUTH
CREATE OR REPLACE FUNCTION public.custom_access_token_hook(event jsonb)
RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
  DECLARE
    claims jsonb;
    user_role text;
    user_org_id uuid;
  BEGIN
    -- Busca os dados oficiais e imutáveis direto da tabela profiles
    SELECT role, organization_id INTO user_role, user_org_id 
    FROM public.profiles 
    WHERE id = (event->>'user_id')::uuid;

    claims := event->'claims';

    -- Se não existir a chave app_metadata, criamos vazia (por segurança)
    IF jsonb_typeof(claims->'app_metadata') IS NULL THEN
      claims := jsonb_set(claims, '{app_metadata}', '{}');
    END IF;

    -- Injetamos a ROLE de forma inviolável no app_metadata do Token
    IF user_role IS NOT NULL THEN
      claims := jsonb_set(claims, '{app_metadata, role}', to_jsonb(user_role));
    END IF;

    -- Injetamos o ORGANIZATION_ID de forma inviolável
    IF user_org_id IS NOT NULL THEN
      claims := jsonb_set(claims, '{app_metadata, organization_id}', to_jsonb(user_org_id));
    END IF;

    -- Atualiza o evento original com as novas claims
    event := jsonb_set(event, '{claims}', claims);
    
    RETURN event;
  END;
$$;

-- 3. PERMISSÕES ESTRITAS PARA O SERVIÇO DE AUTH DO SUPABASE
GRANT EXECUTE ON FUNCTION public.custom_access_token_hook TO supabase_auth_admin;
REVOKE EXECUTE ON FUNCTION public.custom_access_token_hook FROM authenticated, anon, public;
GRANT USAGE ON SCHEMA public TO supabase_auth_admin;

Reasoned
What’s causing operator does not exist: text[] = text
In Supabase Storage, storage.foldername(name) returns an array of text (e.g. ['clientId', 'arquivo...']).
So expressions like:

SQL Query



storage.foldername(name) = auth.uid()::text

try to compare text[] (array) to text (scalar) → hence the error text[] = text.

Supabase docs confirm you should index the array, e.g.:

SQL Query



(storage.foldername(name))[1] = 'private'

Fix (update your policies)
Replace every storage.foldername(name) = auth.uid()::text with:

SQL Query



(storage.foldername(name))[1] = auth.uid()::text

That ensures you compare the first folder segment (the client_id folder) against the current user id as text.

SQL Query



-- ============================================================================
-- STORAGE RLS: bucket 'documents'
-- Caso 1: arquivos ficam em '<client_id>/<arquivo...>'
-- ============================================================================

-- (1) Limpar políticas antigas relacionadas ao bucket documents
DROP POLICY IF EXISTS "Documents viewable by auth users" ON storage.objects;
DROP POLICY IF EXISTS "Documents uploadable by auth users" ON storage.objects;

-- (2) CLIENTE (authenticated) - LEITURA só da própria pasta
CREATE POLICY "documents: client select own folder"
ON storage.objects
FOR SELECT
TO authenticated
USING (
  bucket_id = 'documents'
  AND (storage.foldername(name))[1] = auth.uid()::text
);

-- (3) CLIENTE - INSERÇÃO (upload) só na própria pasta
CREATE POLICY "documents: client insert own folder"
ON storage.objects
FOR INSERT
TO authenticated
WITH CHECK (
  bucket_id = 'documents'
  AND (storage.foldername(name))[1] = auth.uid()::text
);

-- (4) CLIENTE - ATUALIZAÇÃO e DELETE (permitindo dono da pasta)
CREATE POLICY "documents: client update own folder"
ON storage.objects
FOR UPDATE
TO authenticated
USING (
  bucket_id = 'documents'
  AND (storage.foldername(name))[1] = auth.uid()::text
)
WITH CHECK (
  bucket_id = 'documents'
  AND (storage.foldername(name))[1] = auth.uid()::text
);

CREATE POLICY "documents: client delete own folder"
ON storage.objects
FOR DELETE
TO authenticated
USING (
  bucket_id = 'documents'
  AND (storage.foldername(name))[1] = auth.uid()::text
);

-- (5) AGRÔNOMA - VÊ e DELETA somente o que ela criou
CREATE POLICY "documents: agronomist select own uploads"
ON storage.objects
FOR SELECT
TO authenticated
USING (
  bucket_id = 'documents'
  AND owner = auth.uid()
);

CREATE POLICY "documents: agronomist delete own uploads"
ON storage.objects
FOR DELETE
TO authenticated
USING (
  bucket_id = 'documents'
  AND owner = auth.uid()
);

-- (6) Se também quiser que ela ATUALIZE somente o que criou
CREATE POLICY "documents: agronomist update own uploads"
ON storage.objects
FOR UPDATE
TO authenticated
USING (
  bucket_id = 'documents'
  AND owner = auth.uid()
)
WITH CHECK (
  bucket_id = 'documents'
  AND owner = auth.uid()
